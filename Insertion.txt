
.VAR
    value,i,j :INTEGER:=0
    arr : ARRAY_OF INTEGER[];

.BEGIN
    FOR i FROM 0 TO arr.length-1 
	STEP 1  DO
         value := arr[i];
         j = i;
        WHILE (j >= 1 && arr[j - 1] > value) DO
            arr[j] := arr[j - 1];
            j := j - 1;
        END_WHILE
        arr[j] := value;
 
   END_FOR
END
